#在/usr/local/路径下新建mysql目录
cd /usr/local/
mkdir mysql

#切换到mysql路径下，下载或上传安装包

wget https://mirrors.tuna.tsinghua.edu.cn/mysql/downloads/MySQL-8.0/mysql-8.0.29-linux-glibc2.12-x86_64.tar.xz

#xshell使用rzbe+回车上传
rz -be mysql.tar.xz

#解压
tar xvJf mysql-8.0.20-linux-glibc2.12-x86_64.tar.xz 

#在mysql路径下新建data目录
mkdir data

#将解压后的目录重命名
mv mysql-8.0.20-linux-glibc2.12-x86_64 mysql-8.0

#分别创建分组及用户名密码（如已创建就跳过
groupadd mysql
useradd -g mysql mysql

#授权给用户
chown -R mysql.mysql /usr/local/mysql/mysql-8.0
chmod 750 /usr/local/mysql/data -R

#配置环境变量
vim /etc/profile
#在最后添加，保存退出
export PATH=$PATH:/usr/local/mysql/mysql-8.0/bin:/usr/local/mysql/mysql-8.0/lib

#编辑my.cnf文件
vi /etc/my.cnf

# 修改数据库root账号的密码，并设置可以远程访问（因为root默认是不能远程访问的）

use mysql;

#再修改远程连接并生效
update user set host='%' where user='root';

#再刷新
flush privileges;

# 创建新文件夹用以放置datax和dataxweb
cd /home/
mkdir dataXlib
cd dataXlib/
 
 
# 下载解压 datax 
wget https://datax-opensource.oss-cn-hangzhou.aliyuncs.com/20220530/datax.tar.gz
tar xfv datax.tar.gz


#下载解压 datax-web 
wget https://drive.tcosx.com/public/dataX/datax-web-2.1.2.tar.gz
tar xfv datax-web-2.1.2.tar.gz


# 安装 datax-web,按提示要求输入Y,如已安装mysql命令, 则dataxweb会自动配置相应数据库表
cd datax-web-2.1.2
./bin/install.sh
# 如未安装mysql命令，可以取用目录下/bin/db/datax-web.sql脚本去手动执行，完成后修改相关配置文件
vi ./modules/datax-admin/conf/bootstrap.properties
-----------
#Database
#DB_HOST=
#DB_PORT=
#DB_USERNAME=
#DB_PASSWORD=
#DB_DATABASE=
----------


#（可选）配置邮件服务,此文件中包括一些默认配置参数，例如：server.port，具体请查看文件。
 vi ./modules/datax-admin/bin/env.properties
----------
MAIL_USERNAME=""
MAIL_PASSWORD=""
----------

# 配置执行datax的python脚本地址路径 比如（/home/datax/bin/datax.py）到 datax-executor/bin/env.properties
vim modules/datax-executor/bin/env.properties
----------
### 执行datax的python脚本地址
PYTHON_PATH=

### 保持和datax-admin服务的端口一致；默认是9527，如果没改datax-admin的端口，可以忽略
DATAX_ADMIN_PORT=
----------


# Run
./bin/start-all.sh


# jps 出现DataXExecutorApplication和DataXAdminApplication 运行成功
jps
---------
 DataXExecutorApplication
 DataXAdminApplication
---------


# 如果项目启动失败，请检查启动日志：
modules/datax-admin/bin/console.out
# 或者
modules/datax-executor/bin/console.out




# 查看网页  http://{ip}:{port:默认9527}/index.html
# username admin
# pass 123456

# 运行日志
部署完成之后，在modules/对应的项目/data/applogs下(用户也可以自己指定日志，修改application.yml 中的logpath地址即可)，用户可以根据此日志跟踪项目实际启动情况


# 取消所有服务
./bin/stop-all.sh

其他：

项目运行内存最低2G左右，实际使用时会根据数据库表和数据量增加而增加。

DataX迁移PostgreSQL时出现失败的情况

更新 reader 和 writer 中依赖PostgreSQL Jar postgresql-42.3.3.jar 的版本

PostgreSQL JDBC Download

具体的路径在 datax/plugin/reader/postgresqlreader/libs/

具体的路径在 datax/plugin/writer/postgresqlwriter/libs/
